\cventry
{Competitive sailing project for performance optimization and telemetry development}
{Distributed Sensor and Actuation System for SuMoth Sailing Competition}
{PoliMi Sailing Team, Politecnico di Milano, Milan (IT)}
{Oct 2023 - Oct 2024}
{
    Development of marine instrumentation and embedded control systems for competitive sailing electronics. \textbf{Overall Win at 2024 SuMoth competition (1st of 11 teams).}
    \newline
    \vspace{4mm}
    \begin{cvitems}
        \item {Designed sensor acquisition system integrating marine instrumentation using \textbf{NMEA protocol} for real-time environmental data collection.}
        \item {Developed embedded firmware for \textbf{STM32} and \textbf{ESP32} microcontrollers managing sensor acquisition and coordinated actuation via \textbf{CAN bus} communication protocol.}
        \item {Designed CAD models for PCB enclosures and assembled waterproof housings, performing leak testing to ensure marine environmental protection.}
        \item {Implemented wired telemetry and debugging system enabling real-time data logging and performance monitoring during development and testing phases.}
        \item {Contributed to system integration supporting team's extensive on-water validation in challenging marine conditions.}
    \end{cvitems}
}
