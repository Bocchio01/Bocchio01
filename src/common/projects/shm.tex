\cventry
{Academic project for Data Analysis for Mechatronics Systems course}
{SHM as a Multivariate Outlier Detection Problem}
{Politecnico di Milano, Milan (IT)}
{May 2025 - May 2025}
{
    Statistical signal-analysis project for damage detection under environmental variability, separating true anomalies from benign operational effects.
    \newline
    \vspace{4mm}
    \begin{cvitems}
        \item {Analyzed multivariate vibration measurements from tie-rod structures across controlled damage scenarios and varying temperature/humidity conditions.}
        \item {Applied \textbf{Mahalanobis Squared Distance (MSD)} and \textbf{Principal Component Analysis (PCA)} to extract damage-sensitive indicators from sensor data.}
        \item {Defined detection thresholds with statistical control charts and quantified false-alarm behavior under nominal operating conditions.}
        \item {Compared indicator robustness against environmental confounders to improve reliability and uncertainty awareness in diagnostic decisions.}
    \end{cvitems}
}
