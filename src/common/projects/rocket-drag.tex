\cventry
{Academic project for Computational Fluid Dynamics course}
{Drag Coefficient Analysis of Model Rocket Using CFD}
{University of Waterloo, Waterloo (CA)}
{Apr 2024 - Apr 2024}
{
    Computational study to predict aerodynamic performance of a model rocket design, comparing simulation results with theoretical correlations for validation.
    \newline
    \vspace{4mm}
    \begin{cvitems}
        \item {Developed 3D computational domain and structured mesh for rocket geometry using \textbf{Ansys Fluent} preprocessing tools.}
        \item {Performed steady-state \textbf{CFD simulations} across range of Mach numbers, solving compressible Navier-Stokes equations with appropriate turbulence models ($k$-$\omega$ SST).}
        \item {Validated simulation results against analytical drag correlations and empirical data, assessing accuracy for flight trajectory predictions.}
    \end{cvitems}
}
