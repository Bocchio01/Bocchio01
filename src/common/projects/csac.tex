\cventry
{Academic research project for MEMS Technology course}
{Chip Scale Atomic Clocks: Physics and Technology Review}
{University of Waterloo, Waterloo (CA)}
{Mar 2024 - Apr 2024}
{
    Comprehensive study of miniaturized atomic clock technology, examining underlying physics principles and current state-of-the-art implementations for precision timing applications.
    \newline
    \vspace{4mm}
    \begin{cvitems}
        \item {Investigated operating principles of \textbf{coherent population trapping (CPT)} and hyperfine transitions in alkali atoms for frequency stabilization.}
        \item {Analyzed \textbf{MEMS/NEMS} fabrication techniques for vapor cell miniaturization, optical components, and thermal management systems.}
        \item {Compared performance metrics (stability, power consumption, size) of commercial CSAC products against traditional atomic clocks and crystal oscillators.}
        \item {Identified current technological challenges and emerging research directions for improved stability and further miniaturization.}
    \end{cvitems}
}
