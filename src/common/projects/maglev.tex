\cventry
{Academic project for Mechatronics Lab course}
{Modeling and Control of a Magnetic Levitation System}
{Politecnico di Milano, Milan (IT)}
{Sep 2024 - Dec 2024}
{
    End-to-end development of a magnetic levitation experiment, from electromagnetic modeling to real-time control deployment and hardware-based validation.
    \newline
    \vspace{4mm}
    \begin{cvitems}
        \item {Derived a nonlinear electromagnetic model and identified key plant parameters experimentally to obtain a reliable basis for controller and test design.}
        \item {Designed and benchmarked multiple control strategies (\textbf{PID, LQR, MPC}) with state estimation (\textbf{Kalman Filter, Extended Kalman Filter}) for robust feedback from sensor measurements.}
        \item {Built a simulation and analysis workflow in \textbf{MATLAB/Simulink} to validate stability and operating margins across the full 3-22 mm levitation range before hardware tests.}
        \item {Commissioned the control stack on \textbf{RTDAC/PCI I/O hardware} (INTECO), executing repeatable real-time experiments and disturbance-response validation.}
    \end{cvitems}
}
