\cventry
{Academic project for Autonomous Vehicles course}
{Implementation of Path Planning Algorithms for Mobile Robotics}
{Politecnico di Milano, Milan (IT)}
{Apr 2025 - May 2025}
{
    Development and validation of multiple path planning approaches for autonomous navigation, comparing classical graph-based methods with probabilistic sampling techniques.
    \newline
    \vspace{4mm}
    \begin{cvitems}
        \item {Implemented memory-efficient versions of graph search algorithms (\textbf{A*, Dijkstra}) and sample-based planners (\textbf{RRT, RRT*, RRT-Kinodynamic}) in \textbf{MATLAB}.}
        \item {Designed validation framework for benchmarking algorithm performance across 2D and 3D environments with varying obstacle densities.}
        \item {Integrated algorithms with \textbf{ROS ecosystem} for path planning demonstrations on simulated mobile platforms.}
        \item {Analyzed trade-offs between computational efficiency, path optimality, and kinematic feasibility for different planning scenarios.}
    \end{cvitems}
}
