\cventry
{Academic project for Experimental Data Analysis course}
{Visualization of Electronic Density Isosurfaces for Molecular Systems}
{Politecnico di Milano, Milan (IT)}
{Apr 2023 - May 2023}
{
    Development of MATLAB-based tool for processing molecular electronic density data and implementing spatial placement algorithms with geometric constraints.
    \newline
    \vspace{4mm}
    \begin{cvitems}
        \item {Implemented parser for \textbf{.cube file format} to extract pre-computed electronic density data and molecular structure information from quantum chemistry calculations.}
        \item {Generated \textbf{3D isosurface visualizations} and 2D density contour plots using MATLAB graphics, enabling interactive exploration of molecular electron distributions.}
        \item {Implemented file I/O for molecular structure formats, writing \textbf{.xyz output files} compatible with molecular visualization software (VESTA).}
    \end{cvitems}
}
