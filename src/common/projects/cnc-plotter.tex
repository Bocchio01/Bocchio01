\cventry
{Personal project for learning embedded systems and image processing}
{CNC Plotter with Automated Edge Detection}
{Personal Workshop, Como (IT)}
{Jan 2021 - Mar 2021}
{
    Custom-built 2D drawing machine capable of converting digital images into physical sketches, designed as a low-cost learning platform for mechatronics integration.
    \newline
    \vspace{4mm}
    \begin{cvitems}
        \item {Designed and assembled mechanical structure using recycled DVD drive linear actuators and laser-cut wood frame.}
        \item {Implemented \textbf{Arduino-based} motion control firmware with coordinated stepper motor drivers for precise pen positioning.}
        \item {Developed image processing pipeline applying \textbf{Canny edge detection algorithm} to extract drawable contours from photographs.}
        \item {Created path planning software to optimize drawing sequence, minimizing pen lifts and reducing total plotting time.}
    \end{cvitems}
}
