\cventry
{Academic project for Advanced Manufacturing course}
{Thermal Analysis of Laser/Material Interaction in Cutting}
{Politecnico di Milano, Milan (IT)}
{Oct 2023 - Nov 2023}
{
    Theoretical and computational study of energy transfer mechanisms during laser cutting processes, examining vaporization and melt ejection phenomena.
    \newline
    \vspace{4mm}
    \begin{cvitems}
        \item {Developed thermal models for laser energy absorption, heat conduction, and phase transitions (melting, vaporization) in metallic substrates.}
        \item {Analyzed temperature distributions, melt pool dynamics, and kerf geometry as functions of laser parameters (power, scan speed, focal position).}
        \item {Identified optimal processing windows for quality cut production, balancing cut speed against edge quality and heat-affected zone size.}
    \end{cvitems}
}
