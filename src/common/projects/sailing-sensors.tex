\cventry
{Competitive sailing project for performance optimization and telemetry development}
{Distributed Sensor and Actuation System for SuMoth Sailing Competition}
{PoliMi Sailing Team, Politecnico di Milano, Milan (IT)}
{Oct 2023 - Oct 2024}
{
    Development of a distributed instrumentation and embedded control stack for competitive sailing electronics. \textbf{Overall Win at 2024 SuMoth competition (1st of 11 teams).}
    \newline
    \vspace{4mm}
    \begin{cvitems}
        \item {Designed the sensor acquisition architecture and integrated marine instrumentation via \textbf{NMEA protocol} for reliable real-time environmental measurements.}
        \item {Developed embedded firmware on \textbf{STM32} and \textbf{ESP32} for synchronized sensing and coordinated actuation over \textbf{CAN bus}.}
        \item {Produced enclosure CAD and assembled waterproof housings for electronics, including leak-test verification to protect signal integrity in harsh conditions.}
        \item {Implemented wired telemetry and debugging interfaces for live data logging, diagnosis, and iterative test campaigns.}
        \item {Supported full-system integration and commissioning during extensive on-water validation in dynamic operating conditions.}
    \end{cvitems}
}
