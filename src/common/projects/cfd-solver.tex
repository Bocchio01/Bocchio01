\cventry
{Academic project for Computational Fluid Dynamics course}
{Development of 2D CFD Solver for Incompressible Navier-Stokes}
{University of Waterloo, Waterloo (CA)}
{Feb 2024 - Mar 2024}
{
    Implementation of numerical methods for solving fluid flow equations from scratch, developing modular solver architecture for pressure-velocity coupling.
    \newline
    \vspace{4mm}
    \begin{cvitems}
        \item {Implemented pressure-based solution algorithms (\textbf{SCGS and SIMPLE}) in \textbf{C/C++} for 2D incompressible flows on structured grids.}
        \item {Developed finite volume discretization schemes for convective and diffusive terms, ensuring numerical stability and conservation properties.}
        \item {Validated solver accuracy on \textbf{lid-driven cavity benchmark}, comparing velocity profiles and convergence behavior against reference solutions.}
        \item {Analyzed computational performance and iterative convergence characteristics for different relaxation parameters and grid resolutions.}
    \end{cvitems}
}
