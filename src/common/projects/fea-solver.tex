\cventry
{Academic project for Advanced Finite Element Methods course}
{Implementation of Nonlinear FEA Solver with Plasticity}
{University of Waterloo, Waterloo (CA)}
{Jan 2024 - Apr 2024}
{
    Extension of Updated Lagrangian formulation to handle elastic-plastic material behavior, implementing von Mises plasticity with isotropic hardening for metal deformation analysis.
    \newline
    \vspace{4mm}
    \begin{cvitems}
        \item {Implemented \textbf{radial return algorithm} using Newton-Raphson iteration scheme to solve yield condition and compute plastic strain increments in von Mises plasticity framework.}
        \item {Developed consistent tangent operator and stress update procedure for elastic-plastic materials with power-law isotropic hardening model.}
        \item {Compared different objective stress rate formulations (Jaumann vs Truesdell) for elastic constitutive equations in large deformation context.}
    \end{cvitems}
}
