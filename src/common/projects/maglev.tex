\cventry
{Academic project for Mechatronics Lab course}
{Modeling and Control of a Magnetic Levitation System}
{Politecnico di Milano, Milan (IT)}
{Sep 2024 - Dec 2024}
{
    Complete analysis and control design for an unstable magnetic levitation plant, from first-principles modeling through hardware implementation and experimental validation.
    \newline
    \vspace{4mm}
    \begin{cvitems}
        \item {Derived nonlinear dynamical model from electromagnetic principles, including inductance characterization through experimental parameter identification.}
        \item {Designed and compared multiple control strategies (\textbf{PID, LQR, MPC}) coupled with state estimation filters (\textbf{Kalman Filter, Extended Kalman Filter}).}
        \item {Validated controllers through \textbf{MATLAB/Simulink} simulations achieving stability across full operating range (3-22mm levitation height).}
        \item {Deployed control algorithms on \textbf{RTDAC/PCI I/O hardware} from INTECO, demonstrating real-time performance and robustness to disturbances.}
    \end{cvitems}
}
