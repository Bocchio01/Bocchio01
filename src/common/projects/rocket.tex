\cventry
{Personal project to celebrate BSc}
{Model Rocket Design, Simulation, and Flight Test with Onboard Flight Computer}
{Personal Workshop, Como (IT)}
{Jul 2023 - Jul 2023}
{
    Complete design-to-flight project demonstrating integration of simulation-driven optimization, low-cost fabrication, and hardware validation.
    Final design achieved through iterative refinement between \textbf{CAD modeling} and \textbf{CFD simulations}.
    \newline
    \vspace{4mm}
    \begin{cvitems}
        \item {Performed aerodynamic optimization using \textbf{CFD analysis} to minimize drag coefficient, achieving \textbf{$C_D \approx 0.806$} for nose cone and fin geometry.}
        \item {Designed and fabricated rocket structure from low-cost materials (cardboard body, wood fins) with \textbf{3D printed PLA components} for motor mount and recovery system.}
        \item {Developed custom flight computer ($< 20cm^3$) with barometric, temperature, and acceleration sensors for autonomous data logging during flight.}
        \item {Implemented parachute deployment mechanism for controlled descent and safe recovery.}
        \item {Successfully launched and recovered rocket achieving \textbf{538m maximum altitude}, \textbf{120m/s peak velocity}, and \textbf{10g acceleration}, validating simulation predictions.}
    \end{cvitems}
}
