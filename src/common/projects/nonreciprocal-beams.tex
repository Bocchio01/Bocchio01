\cventry
{Academic project for Metamaterials and Metastructures course}
{Study on Nonreciprocal Behavior in Time-Space Modulated Beams}
{Politecnico di Milano, Milan (IT)}
{Nov 2024 - Jan 2025}
{
    Investigation of directional wave propagation in elastic beams controlled by piezoelectric shunts, exploring diode-like behavior for mechanical energy harvesting and isolation applications.
    \newline
    \vspace{4mm}
    \begin{cvitems}
        \item {Performed \textbf{finite element simulations} in \textbf{Comsol Multiphysics} to model coupled electro-mechanical dynamics of time-modulated structures.}
        \item {Designed and analyzed piezoelectric shunt circuits with time-varying impedances to achieve nonreciprocal wave transmission.}
        \item {Processed experimental vibration data in \textbf{MATLAB} to validate simulation predictions and quantify nonreciprocal transmission ratios.}
        \item {Examined frequency-dependent behavior and identified operating ranges for optimal asymmetric wave propagation.}
    \end{cvitems}
}
